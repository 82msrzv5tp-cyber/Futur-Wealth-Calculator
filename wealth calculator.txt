    function weightedReturn(investments) {
      return investments.reduce((sum, item) => sum + (item.allocation / 100) * (item.rate / 100), 0);
    }

    function validateAllocation(investments) {
      const total = investments.reduce((sum, item) => sum + item.allocation, 0);
      const diff = Math.abs(total - 100);
      allocationWarning.textContent = diff > 0.01 ? t("allocationWarning", { total: total.toFixed(1) }) : "";
      return total;
    }

    function readPlan(overrides = {}) {
      const investments = getInvestments();
      const allocationTotal = investments.reduce((sum, item) => sum + item.allocation, 0);
      const baseReturn = allocationTotal ? weightedReturn(investments) * (100 / allocationTotal) : 0;
      return {
        years: Math.min(10, Math.max(1, number("#years"))),
        startingWealth: number("#wealth"),
        monthlyIncome: number("#income"),
        monthlyExpenses: number("#expenses"),
        incomeGrowth: number("#incomeGrowth") / 100,
        inflation: number("#inflation") / 100,
        investPercent: Math.min(100, Math.max(0, number("#investPercent"))) / 100,
        debtBalance: number("#debtBalance"),
        debtInterest: number("#debtInterest") / 100,
        debtPayment: number("#debtPayment"),
        investments,
        allocationTotal,
        effectiveReturn: baseReturn,
        ...overrides
      };
    }

    function runProjection(plan) {
      const {
        years,
        startingWealth,
        monthlyIncome,
        monthlyExpenses,
        incomeGrowth,
        inflation,
        investPercent,
        debtPayment,
        effectiveReturn
      } = plan;

      let wealth = startingWealth;
      let totalContributions = 0;
      const result = [];

      for (let year = 1; year <= years; year++) {
        const annualIncome = monthlyIncome * 12 * Math.pow(1 + incomeGrowth, year - 1);
        const annualExpenses = monthlyExpenses * 12 * Math.pow(1 + inflation, year - 1);
        const annualDebtPayment = debtPayment * 12;
        const availableSavings = Math.max(0, annualIncome - annualExpenses - annualDebtPayment);
        const contribution = availableSavings * investPercent;
        const growth = wealth * effectiveReturn;
        wealth = wealth + growth + contribution;
        totalContributions += contribution;
        const realWealth = wealth / Math.pow(1 + inflation, year);

        result.push({ year, annualIncome, contribution, growth, nominal: wealth, real: realWealth });
      }

      return { result, effectiveReturn, totalContributions, inflation, allocationTotal: plan.allocationTotal, plan };
    }

    function project() {
      const plan = readPlan();
      validateAllocation(plan.investments);
      return runProjection(plan);
    }

    function drawChart(result) {
      const width = chart.width;
      const height = chart.height;
      const pad = 42;
      ctx.clearRect(0, 0, width, height);
      ctx.fillStyle = "#ffffff";
      ctx.fillRect(0, 0, width, height);

      if (!result.length) return;

      const max = Math.max(...result.flatMap(row => [row.nominal, row.real]), 1);
      const x = index => pad + (index / Math.max(1, result.length - 1)) * (width - pad * 2);
      const y = value => height - pad - (value / max) * (height - pad * 2);

      ctx.strokeStyle = "#dfe4dc";
      ctx.lineWidth = 1;
      ctx.font = "12px Inter, sans-serif";
      ctx.fillStyle = "#667064";
      for (let i = 0; i <= 4; i++) {
        const yy = pad + i * ((height - pad * 2) / 4);
        ctx.beginPath();
        ctx.moveTo(pad, yy);
        ctx.lineTo(width - pad, yy);
        ctx.stroke();
      }

      function line(key, color) {
        ctx.strokeStyle = color;
        ctx.lineWidth = 3;
        ctx.beginPath();
        result.forEach((row, index) => {
          const xx = x(index);
          const yy = y(row[key]);
          if (index === 0) ctx.moveTo(xx, yy);
          else ctx.lineTo(xx, yy);
        });
        ctx.stroke();
      }

      line("nominal", "#1f7a4d");
      line("real", "#315f8f");

      result.forEach((row, index) => {
        ctx.fillStyle = "#667064";
        ctx.fillText(String(row.year), x(index) - 3, height - 14);
      });

      ctx.fillStyle = "#1f7a4d";
      ctx.fillRect(width - 210, 18, 14, 14);
      ctx.fillStyle = "#17211b";
      ctx.fillText(t("chartNominal"), width - 188, 30);
      ctx.fillStyle = "#315f8f";
      ctx.fillRect(width - 100, 18, 14, 14);
      ctx.fillStyle = "#17211b";
      ctx.fillText(t("chartReal"), width - 78, 30);
    }

    function drawAllocationChart() {
      const investments = getInvestments().filter(item => item.allocation > 0);
      const width = allocationChart.width;
      const height = allocationChart.height;
      const centerX = 95;
      const centerY = 110;
      const radius = 70;
      const colors = ["#1f7a4d", "#315f8f", "#b66a00", "#7b4ab8", "#c43f3f", "#2f7f7a"];
      const total = investments.reduce((sum, item) => sum + item.allocation, 0) || 1;
      let start = -Math.PI / 2;

      allocationCtx.clearRect(0, 0, width, height);
      allocationCtx.fillStyle = "#fff";
      allocationCtx.fillRect(0, 0, width, height);

      investments.forEach((item, index) => {
        const slice = (item.allocation / total) * Math.PI * 2;
        allocationCtx.beginPath();
        allocationCtx.moveTo(centerX, centerY);
        allocationCtx.arc(centerX, centerY, radius, start, start + slice);
        allocationCtx.closePath();
        allocationCtx.fillStyle = colors[index % colors.length];
        allocationCtx.fill();
        start += slice;
      });

      allocationCtx.font = "12px Inter, sans-serif";
      investments.slice(0, 5).forEach((item, index) => {
        const y = 52 + index * 24;
        allocationCtx.fillStyle = colors[index % colors.length];
        allocationCtx.fillRect(190, y - 10, 12, 12);
        allocationCtx.fillStyle = "#17211b";
        allocationCtx.fillText(`${item.name}: ${pct(item.allocation)}`, 210, y);
      });
    }

    function updateGoal(result) {
      const final = result[result.length - 1] || { real: 0 };
      const goalAmount = number("#goalAmount");
      const goalStatus = document.querySelector("#goalStatus");
      const goalAdvice = document.querySelector("#goalAdvice");

      if (!goalAmount) {
        goalStatus.textContent = money(0);
        goalAdvice.textContent = t("goalNoAmount");
        return;
      }

      const gap = goalAmount - final.real;
      goalStatus.textContent = `${money(final.real)} / ${money(goalAmount)}`;
      goalAdvice.textContent = gap <= 0
        ? t("goalReached", { surplus: money(Math.abs(gap)) })
        : t("goalMissed", { gap: money(gap) });
    }

    function renderScenarios(basePlan) {
      const scenarios = [
        { label: t("scenarioCurrent"), plan: basePlan },
        { label: t("scenarioSaveMore"), plan: { ...basePlan, investPercent: Math.min(1, basePlan.investPercent + 0.1) } },

        { label: t("scenarioInflationLess"), plan: { ...basePlan, inflation: Math.max(0, basePlan.inflation - 0.01) } }
      ];

      document.querySelector("#scenarioGrid").innerHTML = scenarios.map(item => {
        const projection = runProjection(item.plan);
        const final = projection.result[projection.result.length - 1] || { real: 0 };
        return `
          <div class="scenario-card">
            <span>${item.label}</span>
            <strong>${money(final.real)}</strong>
            <p>${t("finalRealValue")}</p>
          </div>
        `;
      }).join("");
    }

      let answer = t("advisorDefaultAnswer");

      if (question.includes("goal") || question.includes("faster") || question.includes("reach")) {
        answer = t("advisorGoalAnswer");
      } else if (question.includes("debt") || question.includes("loan") || debtBalance > 0) {
        answer = t("advisorDebtAnswer");
      } else if (question.includes("risk") || question.includes("portfolio")) {
        answer = t("advisorRiskAnswer");
      }

      document.querySelector("#advisorAnswer").textContent = answer;
    }

    function buildAdvice(effectiveReturn, inflation, allocationTotal) {
      const monthlyIncome = number("#income");
      const monthlyExpenses = number("#expenses");
      const currentWealth = number("#wealth");
      const monthlySavings = monthlyIncome - monthlyExpenses;
      const savingsRate = monthlyIncome > 0 ? monthlySavings / monthlyIncome : 0;
      const emergencyTarget = monthlyExpenses * 6;
      const monthsCovered = monthlyExpenses > 0 ? currentWealth / monthlyExpenses : 0;
      const realReturn = ((1 + effectiveReturn) / (1 + inflation)) - 1;
      const investments = getInvestments();
      const largest = investments.reduce((max, item) => item.allocation > max.allocation ? item : max, { name: "", allocation: 0 });
      const advice = [];
      const riskProfile = document.querySelector("#riskProfile").value;
      const investPercent = Math.min(100, Math.max(0, number("#investPercent")));
      const debtBalance = number("#debtBalance");
      const debtInterest = number("#debtInterest");

      if (monthlySavings <= 0) {
        advice.push(["adviceCashflowBadTitle", "adviceCashflowBadText", {}]);
      } else if (savingsRate < 0.15) {
        advice.push(["adviceCashflowLowTitle", "adviceCashflowLowText", { rate: pct(savingsRate * 100) }]);
      } else {
        advice.push(["adviceCashflowStrongTitle", "adviceCashflowStrongText", { rate: pct(savingsRate * 100) }]);
      }

      if (monthlyExpenses > 0 && currentWealth < emergencyTarget) {
        advice.push(["adviceEmergencyLowTitle", "adviceEmergencyLowText", { target: money(emergencyTarget) }]);
      } else if (monthlyExpenses > 0) {
        advice.push(["adviceEmergencyOkTitle", "adviceEmergencyOkText", {}]);
      }

      if (debtBalance > 0 && debtInterest >= 7) {
        advice.push(["debtWarningTitle", "debtWarningText", { rate: pct(debtInterest) }]);
      }

      if (realReturn < 0.01) {
        advice.push(["adviceReturnLowTitle", "adviceReturnLowText", { rate: pct(realReturn * 100) }]);
      } else {
        advice.push(["adviceReturnOkTitle", "adviceReturnOkText", { rate: pct(realReturn * 100) }]);
      }

      if (largest.allocation > 70) {
        advice.push(["adviceConcentrationTitle", "adviceConcentrationText", {
          name: largest.name,
          allocation: pct(largest.allocation)
        }]);
      }

      if (Math.abs(allocationTotal - 100) > 0.01) {
        advice.push(["adviceAllocationTitle", "adviceAllocationText", { total: allocationTotal.toFixed(1) }]);
      }

      if (investPercent < 100 && monthlySavings > 0) {
        advice.push(["contributionAdviceTitle", "contributionAdviceText", { rate: pct(investPercent) }]);
      }

      if (riskProfile === "conservative") {
        advice.push(["riskConservativeAdviceTitle", "riskConservativeAdviceText", {}]);
      } else if (riskProfile === "aggressive") {
        advice.push(["riskAggressiveAdviceTitle", "riskAggressiveAdviceText", {}]);
      } else {
        advice.push(["riskBalancedAdviceTitle", "riskBalancedAdviceText", {}]);
      }

      advice.push(["adviceFinalTitle", "adviceFinalText", {}]);

      document.querySelector("#advisorSavingsRate").textContent = pct(savingsRate * 100);
      document.querySelector("#advisorRealReturn").textContent = pct(realReturn * 100);
      document.querySelector("#advisorEmergencyFund").textContent = `${money(emergencyTarget)} (${t("monthsCovered", { months: monthsCovered.toFixed(1) })})`;
      document.querySelector("#adviceList").innerHTML = advice.map(([titleKey, textKey, values]) => `
        <li><b>${t(titleKey)}</b> ${t(textKey, values)}</li>
      `).join("");
    }

    function calculate() {
      const { result, effectiveReturn, totalContributions, inflation, allocationTotal } = project();
      const plan = readPlan();
      const final = result[result.length - 1] || { nominal: 0, real: 0 };

      document.querySelector("#finalNominal").textContent = money(final.nominal);
      document.querySelector("#finalReal").textContent = money(final.real);
      document.querySelector("#weightedReturn").textContent = pct(effectiveReturn * 100);
      document.querySelector("#totalContributions").textContent = money(totalContributions);

      rowsEl.innerHTML = result.map(row => `
        <tr>
          <td>${t("yearNumber", { year: row.year })}</td>
          <td>${money(row.annualIncome)}</td>
          <td>${money(row.contribution)}</td>
          <td>${money(row.growth)}</td>
          <td>${money(row.nominal)}</td>
          <td>${money(row.real)}</td>
        </tr>
      `).join("");

      const gap = final.nominal - final.real;
      document.querySelector("#insights").innerHTML = `
        <div><b>${t("inflationImpact")}</b> ${t("inflationImpactText", { inflation: pct(inflation * 100), gap: money(gap) })}</div>
        <div><b>${t("planningNote")}</b> ${t("planningNoteText")}</div>
      `;

      drawChart(result);
      drawAllocationChart();
      updateGoal(result);
      renderScenarios(plan);
      buildAdvice(effectiveReturn, inflation, allocationTotal);
    }

      });
      document.querySelectorAll("[data-i18n-placeholder]").forEach(element => {

      document.querySelectorAll(".icon-btn").forEach(button => button.setAttribute("title", t("removeInvestment")));

      calculate();
    }
